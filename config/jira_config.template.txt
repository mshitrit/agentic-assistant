# ==============================================
# Jira
# ==============================================

# Your Atlassian account email (used for authentication)
JIRA_USER=your-email@redhat.com

# API token for authenticating with Jira Cloud
# Generate one at: https://id.atlassian.com/manage-profile/security/api-tokens
JIRA_TOKEN=your_api_token

# Cloud ID of your Jira instance
# Retrieve it by calling (with your credentials):
# curl -u "your-email:your_token" https://api.atlassian.com/oauth/token/accessible-resources
# Look for the "id" field corresponding to "redhat.atlassian.net"
CLOUD_ID=your-cloud-id

# Comma-separated list of Jira components to watch.
# All open tickets belonging to any listed component will be tracked.
# Ignored if ISSUE_KEY is set in the Debug section.
# Example: COMPONENTS=Storage-based Remediation,Networking
COMPONENTS=ComponentA,ComponentB

# ==============================================
# AI
# ==============================================

# GCP project ID for Claude via Vertex AI
# Enable Claude in Vertex AI Model Garden: https://console.cloud.google.com/vertex-ai/model-garden
# Authenticate locally with: gcloud auth application-default login
GCP_PROJECT_ID=your-gcp-project-id

# GCP region where Claude is available (e.g.global, us-east5, europe-west1)
GCP_REGION=global

# Absolute path to a local clone of the SBR repo.
# Used by the agent to read source files during ticket analysis (tool use).
# Leave empty to disable repo access (agent falls back to memory only).
# Example: SBR_REPO_PATH=/home/user/gitRepos/medik8s/storage-based-remediation
SBR_REPO_PATH=

# Maximum number of SBR repo read operations (read_file + list_directory) per ticket analysis.
# Prevents excessive file reads and cost spikes. Default: 10.
MAX_READ_CALLS=10

# Maximum number of living memory write operations per ticket analysis.
# Kept separate so memory updates are never blocked by read budget exhaustion. Default: 3.
MAX_WRITE_CALLS=3

# ==============================================
# Debug
# ==============================================

# Track a specific ticket — takes priority over COMPONENTS if set (e.g. PROJ-123)
# Leave empty or remove to use COMPONENTS instead
ISSUE_KEY=PROJ-123

# Log level — controls verbosity of console output
# INFO:  standard output (default)
# DEBUG: also prints the full agent prompt before each Claude call
LOG_LEVEL=INFO

# Debug mode — controls which external calls are active
# PRODUCTION:   full behaviour, posts to Jira and calls Claude (default)
# DISABLE_JIRA: print comments to console instead of posting to Jira
# DISABLE_AI:   skip Claude call, use a hardcoded response instead
# FULL_DISABLE: combine DISABLE_JIRA + DISABLE_AI (no external calls at all)
DEBUG_MODE=PRODUCTION
